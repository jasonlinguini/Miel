function drawBee(b){ X.save(); X.translate(b.x, b.y); X.rotate(Math.atan2(b.vy, b.vx)); // corps X.fillStyle = (b.t === 'Q') ? '#ffe08f' : '#e7c16a'; X.beginPath(); X.ellipse(0, 0, b.r + 2, b.r, 0, 0, Math.PI * 2); X.fill(); // rayures X.lineWidth = 2; X.strokeStyle = '#2b2418'; X.beginPath(); X.moveTo(-b.r, -2); X.lineTo(b.r, -2); X.moveTo(-b.r, 2); X.lineTo(b.r, 2); X.stroke(); X.restore(); }
